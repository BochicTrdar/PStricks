\def\setCnt#1.#2.#3\@nil{\pst@cntn=#1}%
\def\psplaceChar{\@ifnextchar[\psplaceChar@i{\psplaceChar@i[\huge]}}
\def\psplaceChar@i[#1](#2)#3#4{%
  \pst@cnta=0\relax
  \pssetlength{\pst@dima}{#3}%
  \expandafter\pscountChars#4;%
  \pstFPDiv\pst@tempa{360}{\the\pst@cnta}%
  \pstFPmul\pst@tempb{\strip@pt\pst@dima}{\psPiTwo}%
  \pstFPdiv\pst@tempb{\pst@tempb}{\the\pst@cnta}%
  \expandafter\setCnt\pst@tempa..\@nil%
  \def\pst@Radius{#3}%
  \pst@cntm=0\relax
  \ifx*#1
    \pst@dimo=\pst@tempb pt
    \def\@fontsize{\fontsize{\pst@dimo}{1.01\pst@dimo}\selectfont}
  \else\def\@fontsize{#1}\fi
  \rput(#2){\expandafter\psplaceChar@ii#4;}
}
\def\pscountChars#1#2;{ %
  \advance\pst@cnta by \@ne
  \ifx\relax#2\relax\else\expandafter\pscountChars#2;\fi%
}
\def\psplaceChar@ii#1#2;{%
  \uput[\the\pst@cntm]{!180 \the\pst@cntm\space add}(\pst@Radius;\the\pst@cntm){\@fontsize#1}%
  \advance\pst@cntm by \pst@cntn
  \ifx\relax#2\relax\else\expandafter\psplaceChar@ii#2;\fi%
}
