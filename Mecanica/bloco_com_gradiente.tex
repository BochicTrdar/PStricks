\usepackage{pst-grad} % For gradients

  \psframe[linestyle=none,fillstyle=gradient,gradmidpoint=0.5,
           gradbegin=red,gradend=blue,gradlines=200](-2.5,-1.5)(2.5,1.5)

  \psline[linecolor=gray,linestyle=dashed](-2.5,0)(2.5,0) % horizontal center line
  \psline[linecolor=gray,linestyle=dashed](0,-1.5)(0,1.5) % vertical center line
  \psdot[dotsize=5pt,linecolor=black](0,0) % center point
\end{pspicture}
