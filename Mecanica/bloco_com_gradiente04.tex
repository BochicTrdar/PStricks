\documentclass{article}

\pagestyle{empty}

\usepackage{pstricks}
\usepackage{pst-grad}

% Minimal gradient box command
\newcommand{\Gbox}[6]{%
  % #1 = center x
  % #2 = center y
  % #3 = width
  % #4 = height
  % #5 = start color
  % #6 = end color
  \psframe[linestyle=none,fillstyle=gradient,gradmidpoint=0.5,
           gradbegin=#5,gradend=#6,gradlines=150]
          (!#1 #3 2 div sub #2 #4 2 div sub)
          (!#1 #3 2 div add #2 #4 2 div add)
}

\begin{document}

\begin{pspicture}(-4,-3)(4,3)
  % Place boxes wherever you want
  \Gbox{0}{0}{3}{2}{red}{blue}
  \Gbox{-2}{1.5}{1.5}{1}{green!70!black}{yellow}
  \Gbox{2.5}{-1}{2}{1}{purple}{white}
  \Gbox{0}{-2}{1}{1}{black}{gray!50}
\end{pspicture}

\end{document}
