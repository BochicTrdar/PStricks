\documentclass{article}

\pagestyle{empty}

\usepackage{pstricks}
\usepackage{pst-grad}
\usepackage{multido} % For repeating patterns

% Define a custom command for gradient blocks
\newcommand{\gradientbox}[5][]{%
  % #1 = optional additional parameters
  % #2 = x position (center)
  % #3 = y position (center)
  % #4 = width
  % #5 = height
  \psframe[%
    linestyle=none,
    fillstyle=gradient,
    gradmidpoint=0.5,
    gradbegin=#1,
    gradend=blue!80!black,
    gradlines=150
  ]([nodesep=#4,angle=180]#2,#3)([nodesep=#4,angle=0]#2,#3)%
  ([nodesep=#5,angle=90]#2,#3)([nodesep=#5,angle=-90]#2,#3)
}

% Alternative simpler version with corner coordinates
\newcommand{\GradientBlock}[6][]{%
  % #1 = optional parameters (colors, etc.)
  % #2 = bottom-left x
  % #3 = bottom-left y
  % #4 = width
  % #5 = height
  % #6 = color pair (e.g., "red,blue")
  \begingroup
    \pgfkeys{/grad/.is family, /grad,
      begin color/.initial=red,
      end color/.initial=blue,
      lines/.initial=150
    }
    \pgfqkeys{/grad}{#1}%
    \edef\gradbegin{\pgfkeysvalueof{/grad/begin color}}
    \edef\gradend{\pgfkeysvalueof{/grad/end color}}
    \edef\gradlines{\pgfkeysvalueof{/grad/lines}}
    \psframe[linestyle=none,fillstyle=gradient,
             gradmidpoint=0.5,gradbegin=\gradbegin,
             gradend=\gradend,gradlines=\gradlines]
            (#2,#3)(!#2 #4 add #3 #5 add)
  \endgroup
}

% Even simpler version with center-based coordinates
\newcommand{\CGradientBox}[5][]{%
  % #1 = optional style parameters
  % #2 = center x
  % #3 = center y
  % #4 = total width
  % #5 = total height
  \begingroup
    \pgfkeys{/gbox/.is family, /gbox,
      color1/.initial=red,
      color2/.initial=blue,
      lines/.initial=150
    }
    \pgfqkeys{/gbox}{#1}%
    \edef\colorone{\pgfkeysvalueof{/gbox/color1}}
    \edef\colortwo{\pgfkeysvalueof{/gbox/color2}}
    \edef\gradlines{\pgfkeysvalueof{/gbox/lines}}
    \psframe[linestyle=none,fillstyle=gradient,gradmidpoint=0.5,
             gradbegin=\colorone,gradend=\colortwo,gradlines=\gradlines]
            (!#2 #4 2 div sub #3 #5 2 div sub)
            (!#2 #4 2 div add #3 #5 2 div add)
  \endgroup
}

\begin{document}

\begin{pspicture}(-5,-3)(5,3)
  % Example 1: Using CGradientBox (center-based, simplest)
  \CGradientBox[color1=yellow,color2=red]{0}{0}{4}{2}
  
  % Example 2: Multiple boxes with different colors
  \CGradientBox[color1=green!80!black,color2=white]{-3}{2}{2}{1}
  \CGradientBox[color1=white,color2=blue]{3}{2}{2}{1}
  \CGradientBox[color1=purple,color2=yellow]{3}{-2}{1.5}{1.5}
  
  % Example 3: Using GradientBlock (corner-based)
  \GradientBlock[begin color=orange,end color=cyan]{-4}{-2}{2}{1.5}
  
  % Example 4: Create a grid of gradient boxes
  \multido{\iX=-2+1,\iA=10+15}{5}{%
    \multido{\iY=-1+1,\iB=30+10}{3}{%
      \CGradientBox[color1=blue!\iA!green,color2=red!\iB!yellow]{\iX}{\iY}{0.8}{0.8}
    }
  }
  
  % Optional: Add coordinate labels
  %\psaxes[labels=none,ticks=none,linecolor=gray!30](0,0)(-5,-3)(5,3)
  \rput(0,0){\small (0,0)}
\end{pspicture}

\end{document}
